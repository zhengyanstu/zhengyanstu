#!/usr/bin/env python
# coding: utf-8
# usage: python script.py input.fa > output.axt
import sys
def converter(FileName):
	Seq = {}; id_list = []
	id = ''
	with open(FileName, 'r') as fr:
		for line in fr.readlines():
			if line[0] == '>':
				id = line[1:].rstrip()
				#id = header.split("_")[0]
				id_list.append(id)
				Seq[id] = ''
			else:
				Seq[id] += line.rstrip()
	return Seq, id_list
Seq, id_list = converter(sys.argv[1])
header = "-".join(id_list)
print(header)
newseq = []
for id in id_list:
	newseq.append(Seq[id])
newseq = "\n".join(newseq)
print(newseq)
